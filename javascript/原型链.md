# Javascript 原型链

```javascript
const f = function () {
   this.a = 1;
   this.b = 2;
}

f.prototype.b = 3;
f.prototype.c = 4;

const obj = new f();

console.log(f);
console.log(obj);
// {a:1, b:2} ---> {b:3, c:4} ---> Object.prototype---> null
```
